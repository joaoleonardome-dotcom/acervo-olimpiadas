Codigo para o Google Apps Script:
function doGet(e){
  var action = e&&e.parameter&&e.parameter.action;
  var sheet = getSheet();
  if(action==='load'){
    var val = sheet.getRange(2,1).getValue();
    return out(val||'{"olimpiadas":[],"provas":[]}');
  }
  return out('{"olimpiadas":[],"provas":[]}');
}
function doPost(e){
  var sheet = getSheet();
  sheet.getRange(2,1).setValue(e.postData.contents);
  return out('ok');
}
function getSheet(){
  var p=PropertiesService.getScriptProperties().getProperty('SID');
  var ss=p?SpreadsheetApp.openById(p):SpreadsheetApp.create('AcervoOlimpiadas');
  if(!p){PropertiesService.getScriptProperties().setProperty('SID',ss.getId());}
  var sh=ss.getSheetByName('data');
  if(!sh){sh=ss.insertSheet('data');sh.appendRow(['json']);}
  return sh;
}
function out(s){
  return ContentService.createTextOutput(s).setMimeType(ContentService.MimeType.JSON);
}
